cmake_minimum_required(VERSION 3.5...3.16)

project(xmq_host_service VERSION "${BUILD_VERSION_STRING}" LANGUAGES CXX)

file(GLOB_RECURSE src "*.cpp")
file(GLOB_RECURSE framework_network_src "${SOURCE_DIR}/network/xmq/*.cpp")
file(GLOB_RECURSE framework_utils_src "${SOURCE_DIR}/utils/*.cpp")

include_directories(${INCLUDE_DIR})
link_directories(${LINK_DIR}/liblog)
link_directories(${LINK_DIR}/libxmq)

add_executable(
	${PROJECT_NAME} 
	${src} 
	${framework_network_src} 
	${framework_utils_src})
target_link_libraries(
	${PROJECT_NAME} 
	PRIVATE 
	xmq log 
	libboost_program_options.a 
	libboost_thread.a 
	pthread)
